rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() {
      return request.auth != null;
    }

    function currentUserDocPath() {
      return /databases/$(database)/documents/artifacts/TrinityCart-default-app-id/users/$(request.auth.uid);
    }

    function isActiveUser() {
      return isSignedIn()
        && exists(currentUserDocPath())
        && get(currentUserDocPath()).data.isActive == true;
    }

    function isAdmin() {
      return isActiveUser()
        && get(currentUserDocPath()).data.role == "admin";
    }

    function isSelf(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    function isValidSelfGuestBootstrap(userId) {
      return isSelf(userId)
        && request.resource.data.UID == request.auth.uid
        && request.resource.data.email == request.auth.token.email
        && request.resource.data.role == "guest"
        && request.resource.data.isActive == true
        && request.resource.data.createdBy == request.auth.token.email
        && request.resource.data.updatedBy == request.auth.token.email;
    }

    // Users:
    // - any signed-in user can read only their own user doc
    // - admin can list and manage all users
    // - first-time sign in can create a guest record for self
    match /artifacts/TrinityCart-default-app-id/users/{userId} {
      allow get: if isSelf(userId) || isAdmin();
      allow list: if isAdmin();
      allow create: if isAdmin() || isValidSelfGuestBootstrap(userId);
      allow update, delete: if isAdmin();
    }

    // Admin master data:
    // keep reads open to active users so both TrinityCart and Moneta continue working,
    // but restrict writes to administrators.
    match /artifacts/TrinityCart-default-app-id/productCategories/{docId} {
      allow read: if isActiveUser();
      allow write: if isAdmin();
    }

    match /artifacts/TrinityCart-default-app-id/paymentModes/{docId} {
      allow read: if isActiveUser();
      allow write: if isAdmin();
    }

    match /artifacts/TrinityCart-default-app-id/salesSeasons/{docId} {
      allow read: if isActiveUser();
      allow write: if isAdmin();
    }

    match /artifacts/TrinityCart-default-app-id/saleTypes/{docId} {
      allow read: if isActiveUser();
      allow write: if isAdmin();
    }

    match /artifacts/TrinityCart-default-app-id/salesEvents/{docId} {
      allow read: if isActiveUser();
      allow write: if isAdmin();
    }

    // Compatibility-first operational data rules for the existing TrinityCart app
    // and the Moneta rebuild. These preserve the current behavior for authenticated,
    // active users so the legacy app should keep working.
    match /artifacts/TrinityCart-default-app-id/suppliers/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/products/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/productCatalogue/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/purchaseInvoices/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/inventoryLedger/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/supplierPaymentsLedger/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/salesCatalogues/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/draftSales/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/churchTeams/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/userTeamMemberships/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/consignmentOrders/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/consignmentPaymentsLedger/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/systemSetups/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/salesPaymentsLedger/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/salesInvoices/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/donations/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/expenses/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/leads/{document=**} {
      allow read, write: if isActiveUser();
    }

    match /artifacts/TrinityCart-default-app-id/consignmentOrdersV2/{document=**} {
      allow read, write: if isActiveUser();
    }

    // Required for collectionGroup("activityLog") queries used by reports.
    match /{path=**}/activityLog/{activityId} {
      allow read: if isActiveUser();
    }
  }
}
